}

04. 배열 : 데이터 불러오기 : 2차 배열

객체 속의 변수 생성 가능 

{
    const arr = [100, 200, ["javascript","jquery"]]     //arr 배열에 100,200,세번째 자리에 배열 삽입 후 javascript,jquery 삽입

    document.write(arr[0]);       //arr 배열 첫번째에 100 대입
    document.write(arr[1]);       //arr 배열 두번째에 200 대입
    document.write(arr[2][0]);       //arr에 배열 세번째의 배열에 첫번째 배열에 javascript 대입
    document.write(arr[2][1]);       //arr에 배열 세번째의 배열에 첫번째 배열에 jquery 대입
}

06. 배열 : 데이터 불러오기 : for()문

객체 속의 변수 생성 가능 

{
    const arr = [100, 200, 300, 400, 500, 600, 700, 800, 900];       //arr 배열 100~900까지 순서대로 대입

    document.write(arr[0]);       //arr 배열 100 출력
    document.write(arr[1]);       //arr 배열 200 출력
    document.write(arr[2]);       //arr 배열 300 출력
    document.write(arr[3]);       //arr 배열 400 출력
    document.write(arr[4]);       //arr 배열 500 출력
    document.write(arr[5]);       //arr 배열 600 출력
    document.write(arr[6]);       //arr 배열 700 출력
    document.write(arr[7]);       //arr 배열 800 출력
    document.write(arr[8]);       //arr 배열 800 출력

    for(초기값, 조건식, 증감식)
    for(let i=0; i<arr.length; i++){     //for(초기값, 조건식 배열변수.length, 증감식)
        document.write(arr[i], "<br>");
    }
}

07. 배열 : 데이터 불러오기 : forEach()

객체 속의 변수 생성 가능 

{
    const num = [100, 200, 300, 400, 500];
            // 첫번째 방법
            for(let i=0; i<5; i++){
                document.write(num[i],"<br>");
            }

            document.write("<br>
");

            // 두번째 방법
            for(let i=0; i<num.length; i++){
                document.write(num[i],"<br>");
            }

            // 세번째 방법
            //forEach()를 이용해서 출력
            num.forEach(function(el){//function은 콜백함수 : 첫번째 함수가 끝나면 두번째 함수가 시작되는걸 의미
                document.write(el);
            });

            document.write("<br><br>");

            //forEach() 3가지 인자()값
            num.forEach(function(element, index, array){
                document.write(element, "<br>"); // 각 저장된 값
                document.write(index, "<br>"); //각 자리의 위치
                document.write(array, "<br>"); // 각 저장된 값을 쉼표까지
            });
}
