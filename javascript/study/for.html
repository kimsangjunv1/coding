for문

for문은 특정 구문을 여러번 반복하고 싶을 때 사용하는 반복문입니다.

for ( 초기값; 조건식; 증감값 ){
    실행문
}

0부터 99까지 출력하기

//0부터 99까지 출력하기
for( let i = 0; i<100; i++){       //i의 값을 1씩 증가
    document.write(i)       //i 출력
}

1부터 100까지 출력하기

//1부터 100까지 출력하기
    for( let i = 0; i<=100; i++){       //i의 값을 1씩 증가
document.write(i)       //i 출력
}

1부터 100까지 출력하기(짝수)

짝수로 출력할때는 증가값을 2로 해주면 됩니다.

//1부터 100까지 출력하기
for( let i = 2; i<=100; i+=2){        //i의 값을 100까지 2씩 더하기
document.write(i)       //i 출력
}

for문을 이용해서 테이블 출력하기

중첩 for문을 이용해서 테이블을 만들 수 있습니다.

//1부터 100까지 출력하기
let table = "<table>";        //변수 테이블 값 <table> 생성
    
    for (let i=0; i<10; i++){        //i의 값이 10이되기 전까지 1씩 더하며 반복
        
        table += "<tr>";      //테이블에 tr 더하기

        for(let j=0; j<10; j++){        //j의 값이 10이되기 전까지 1씩 더하며 반복
            table += "<td>1</td>";      //테이블에 td 더하기
        }
        table += "</tr>"      //tr 닫기태그 더하기
    
    table += "</table>"      //table 닫기태그 더하기

document.write(table);      //테이블 출력
